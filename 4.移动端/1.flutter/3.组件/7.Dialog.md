## 7.Dialog

- AlertDialog
- SimpleDialog
- showModalButtonSheet
- 关闭Dialog
  - Navigator.of(context).pop(): 关闭Dialog
  - Navigator.pop(context, "中文"); 关闭Dialog
  - result: Navigator.of(context).pop("ok") 返回的值

### 7.1 AlertDialog

```dart
var result = await showDialog(
    barrierDismissible: false, // 点击蒙层消失 默认true消失
    context: context,
    builder: (context) {
      return AlertDialog(
        title: Text("Alert"),
        content: Text("This is an alert dialog"),
        actions: [
          TextButton(
              onPressed: () =>
                  {print("确定"), Navigator.of(context).pop("ok")},
              child: Text("确定")),
          TextButton(
              onPressed: () => Navigator.of(context).pop("cancel"),
              child: Text("取消"))
        ],
      );
    });
print(result);
```

### 7.2 SimpleDialog

```dart
var result = await showDialog(
    barrierDismissible: false, // 点击蒙层消失 默认true消失
    context: context,
    builder: (context) {
      return SimpleDialog(
        title: Text("SimpleDialog title"),
        children: <Widget>[
          SimpleDialogOption(
              child: Text("中文"),
              onPressed: () {
                // Navigator.of(context).pop("中文");
                Navigator.pop(context, "中文");
              }),
          SimpleDialogOption(
              child: Text("英文"),
              onPressed: () {
                Navigator.of(context).pop("英文");
              }),
          SimpleDialogOption(
              child: Text("法文"),
              onPressed: () {
                Navigator.of(context).pop("法文");
              }),
        ],
      );
    });
print(result);
```

### 7.3 自定义Dialog

```dart

class CustomDialog extends StatefulWidget {
  final String title;
  final String content;
  final Function()? onClose;
  const CustomDialog(
      {super.key, this.title = "tip", this.content = "content", this.onClose});
  @override
  _CustomDialogState createState() {
    return _CustomDialogState();
  }
}

class _CustomDialogState extends State<CustomDialog> {
  @override
  Widget build(BuildContext context) {
    return Material(
      // 透明背景
      type: MaterialType.transparency,
      // 包裹组件,否则会全屏
      child: Center(
        child: Container(
          width: 300,
          height: 300,
          color: Colors.white,
          child: Column(
            children: [
              Padding(
                padding: EdgeInsets.all(10),
                child: Stack(
                  children: [
                    Align(
                      alignment: Alignment.centerLeft,
                      child: Text(
                        widget.title,
                        style: TextStyle(fontSize: 20),
                      ),
                    ),
                    Align(
                      alignment: Alignment.centerRight,
                      child: InkWell(
                          onTap: () {
                            if (widget.onClose != null) {
                              widget.onClose!();
                            } else {
                              Navigator.pop(context);
                            }
                          },
                          child: Icon(Icons.close)),
                    ),
                  ],
                ),
              ),
              Divider(),
              Padding(
                padding: EdgeInsets.all(10),
                child: SizedBox(
                  width: double.infinity,
                  child: Text(widget.content),
                ),
              )
            ],
          ),
        ),
      ),
    );
  }
}
// 使用
showDialog(
  context: context,
  builder: (context) {
    return CustomDialog(
      title: "提示",
      content: "内容",
    );
  });
```
